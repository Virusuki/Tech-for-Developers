# 그래프 - 가장 먼 노드 (프로그래머스)

문제 - https://programmers.co.kr/learn/courses/30/lessons/49189?language=python3


from collections import deque

def solution(n, edge):
    edj = [[] for _ in range(n+1)]
    distance = [-1] * (n+1)    # distance에 -1은 시작점은 cost를 0으로 초기화 하기 위해서,,,
    q = deque([1])
    distance[1] = 0
    
    for e in edge:
        edj[e[0]].append(e[1])
        edj[e[1]].append(e[0])
    
    while q:
        cur = q.popleft()
        
        for i in edj[cur]:
            if distance[i] == -1:
                q.append(i)
                distance[i] = distance[cur]+1    # 현재 distance의 가중치에서 다음 노드에 +1 증가
    cnt=0
    for i in distance:
        if max(distance) == i:
            cnt += 1
            
    return cnt

edge = [[3, 6], [4, 3], [3, 2], [1, 3], [1, 2], [2, 4], [5, 2]]
n = 6

res = solution(n, edge)

print(res)
