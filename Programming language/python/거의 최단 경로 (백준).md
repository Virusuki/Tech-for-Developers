# 거의 최단 경로 (백준)

문제 - https://www.acmicpc.net/problem/5719

```
import heapq
N, M = map(int, input().split())

INF = float('INF')
start = 0
distance = [INF] * (N+1)

adj = [[] for _ in range(N+1)]
for _ in range(M):
    x, y, c = map(int, input().split())
    adj[x].append((y,c))
    

def dijkstra(start):
    q = []
    
    heapq.heappush(q, (start,0))  # node , w
    distance[start] = 0
    
    while q:
        cur_node, dist = heappop(q)
        
        if distance[cur_node] < dist:
            continue
                
        for next, w in adj[cur_node]:
            total_w = dist + w
            if total_w < distance[next]:
                distance[next] = total_w
                heapq.heappush(q, (next, total_w))

dijkstra(start)
print(distance)
```
