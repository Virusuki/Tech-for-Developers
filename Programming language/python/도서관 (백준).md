# 도서관 (백준)

문제 - https://www.acmicpc.net/problem/1461

```
import heapq

n = [-37, 2, -6, -39, -29, 11, -28]
q=[]

positive = []
negative = []


for i in n:
    if i > 0:
        heapq.heappush(positive, -i)
    else:
        heapq.heappush(negative, i)

print(positive)
print(negative)

result = 0 
while negative:
    result += heapq.heappop(negative)
    
    for _ in range(2-1): # 2 is m
        if negative:
            heapq.heappop(negative)
print(result)
```
