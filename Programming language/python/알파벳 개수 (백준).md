    'e': 5,

    'g': 7,
    'h': 8,

answer = [0] * (len(alphabet)+1)

for i in s:
    k=0
    k=alphabet[i]
    answer[k] += 1
for k in range(1, len(answer)):
    print(answer[k], end=' ')
