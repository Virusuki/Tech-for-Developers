s = "{{20,111},{111}}"	

s = s.replace('{', '')
s = s.split('}')
num = []
answer = []
print(s)
for i in range(len(s)-1, -1, -1):
	if s[i] == '':
		s.pop(i)
	else:
		s[i] = s[i].split(',')
		if i:
			s[i].pop(0)
		s[i] = list(map(int, s[i]))
s = sorted(s, key=lambda x: len(x))

for k in s:
	for j in k:
		if j not in num:
				num.append(j)
print(num)
