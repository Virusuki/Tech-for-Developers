# 올바른 괄호
- stack을 이용한 괄호가 바르게 짝지어졌는지 확인

```
def solution(Parenthesis):
    stack = []
    
    for ch in Parenthesis:
        if ch == '(':  
            stack.append(ch)
        else:                  # if ')' do pop
            if len(stack):
                stack.pop()
            else:
                return False            
        
    return False if len(stack) else True
    
        
Parenthesis = "((())())"

ans = solution(Parenthesis)

print(ans)
```
