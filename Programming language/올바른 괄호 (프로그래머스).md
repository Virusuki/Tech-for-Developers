def solution(Parenthesis):
    stack = []
    
    for ch in Parenthesis:
        if ch == '(':  
            stack.append(ch)
        else:                  # if ')' do pop
            if len(stack):
                stack.pop()
            else:
                return False            
        
    return False if len(stack) else True
    
        
Parenthesis = "((())())"

ans = solution(Parenthesis)

print(ans)
